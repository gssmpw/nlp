\NewDocumentCommand{\dcurl}{sO{}m}{%
  \IfBooleanTF{#1}
    {\dcurlext{#3}}
    {\dcurlx[#2]{#3}}%
}
\NewDocumentCommand{\dcurlext}{m}{%
  \sbox0{%
    \mathsurround=0pt % just for safety
    $\left\{\vphantom{#1}\right.\kern-\nulldelimiterspace$%
  }%
  \sbox2{\{}%
  \ifdim\ht0=\ht2
    \{\kern-.625\wd2 \{#1\}\kern-.625\wd2 \}%
  \else
    \left\{\kern-.7\wd0\left\{#1\right\}\kern-.7\wd0\right\}%
  \fi
}
\NewDocumentCommand{\dcurlx}{om}{%
  \sbox0{\mathsurround=0pt$#1\{$}%
  \sbox2{\{}%
  \ifdim\ht0=\ht2
    \{\kern-.625\wd2 \{#2\}\kern-.625\wd2 \}%
  \else
    \mathopen{#1\{\kern-.7\wd0 #1\{}
    #2
    \mathclose{#1\}\kern-.7\wd0 #1\}}
  \fi
}

\NewDocumentCommand{\dsquare}{sO{}m}{%
  \IfBooleanTF{#1}
    {\dsquareext{#3}}
    {\dsquarex[#2]{#3}}%
}
\NewDocumentCommand{\dsquareext}{m}{%
  \sbox0{%
    \mathsurround=0pt % just for safety
    $\left[\vphantom{#1}\right.\kern-\nulldelimiterspace$%
  }%
  \sbox2{[}%
  \ifdim\ht0=\ht2
    [\kern-.525\wd2 [#1]\kern-.525\wd2 ]%
  \else
    \left[\kern-.6\wd0\left[#1\right]\kern-.6\wd0\right]%
  \fi
}
\NewDocumentCommand{\dsquarex}{om}{%
  \sbox0{\mathsurround=0pt$#1[$}%
  \sbox2{[}%
  \ifdim\ht0=\ht2
    [\kern-.525\wd2 [#2]\kern-.525\wd2 ]%
  \else
    \mathopen{#1[\kern-.6\wd0 #1[}
    #2
    \mathclose{#1]\kern-.6\wd0 #1]}
  \fi
}
