\begin{algorithm}
\begin{algorithmic}
\STATE {\bf Participants}: Players $P_0, P_1$
\STATE {\bf Input}: Merkle tree roots $\rt_0, \rt_1$
\STATE
\STATE $(\currt_0, \currt_1) := (\rt_0, \rt_1)$
\STATE $(\start, \sfend) = (0, n)$
\FOR{$i = 0$ to $\log n$}
    \STATE $P_1, P_2$ send $(l_1, r_1), (l_2, r_2)$, respectively.
    \STATE If $H(l_i, r_i) \neq \currt_i$ for either $i$, then $R$ eliminates $P_i$
    \IF{$l_1 = l_2$}
        \STATE $(\currt_0, \currt_1) = (r_0, r_1)$
        \STATE $\start = (\start+\sfend)/2$
    \ELSE{}
        \STATE $(\currt_0, \currt_1) = (l_0, l_1)$
        \STATE $\sfend = (\start+\sfend)/2$
    \ENDIF
\ENDFOR
\STATE Let $\displeaves = (\currt_0, \currt_1)$ 
\STATE Disputed index $\dispidx = \start$
\STATE Referee outputs: $(\dispidx, \displeaves)$
\STATE
\end{algorithmic}
\caption{{\bf Dispute bisection game.} The players and the referee 
identify the first step the two players diverged in their computation.}
\label{alg:protocol_bisection_background}
\end{algorithm}
