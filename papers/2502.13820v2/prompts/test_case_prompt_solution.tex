from typing import Optional
def first_repeated_char(s: str) -> Optional[str]:
    """ 
    Find the first repeated character in a given string.
    >>> first_repeated_char("abbac")
    'a'
    """
    for index, c in enumerate(s):
        if s[:index + 1].count(c) > 1:
            return c
    return None
