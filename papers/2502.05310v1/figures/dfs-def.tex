dfs :: SearchPolicy '[Branch, Fail]
dfs (Success _ x) = yield x
dfs (SomeNode _ (Z (Node (Branch cands) child))) =
    streamBind (getStream cands) (dfs . child)
dfs (SomeNode _ (S (Z (Node Fail _)))) = mzero
