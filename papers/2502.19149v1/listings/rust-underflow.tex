  pub fn maximum_points(mut arr: Vec<i32>, mut curr: i32) -> i64 {
    arr.sort_unstable();
    let mut res = 0;
    let mut pos = arr.len() - 1;

    while pos <= arr.len() && arr[0] <= curr {
      res += (curr / arr[0]) as i64;
      curr %
      curr += arr[pos];
      pos -= 1;
    }

    return res;
  }
