function maximum_points(arr, curr)
    sort arr in ascending order
    res = 0
    pos = arr.len() - 1

    while pos >= 0 and arr[0] <= curr:
        res += curr // arr[0]
        curr = curr %
        curr += arr[pos]
        pos -= 1

    return res
