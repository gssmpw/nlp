\text{SynMM}(\bar{\vv}_1, \bar{\vv}_2) = \text{Linear}\left(\bar{\vv}_1 \cat \bar{\vv}_2\right),
\end{equation}

\begin{equation}
    \bar{\vv}_1 = \text{Max}(V) = \max _{\vv_i \in V} \vv_i,
\end{equation}

     \bar{\vv}_2 = \text{Mean}(V) = \frac{1}{N^v} \sum_{\vv_i \in V} \vv_i.
\end{equation}
