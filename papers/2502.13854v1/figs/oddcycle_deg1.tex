\documentclass[tikz]{standalone}

\usepackage{xcolor}

\definecolor{stgoblue}{RGB}{74,144,226}
\definecolor{stgogreen}{RGB}{80,227,194}
\definecolor{stgored}{RGB}{255,69,0}
\definecolor{stgoorange}{RGB}{255,165,0}

\usetikzlibrary{fit,shapes.geometric,positioning,calc}

\begin{document}

\begin{tikzpicture}
    % Node styles
    \colorlet{colorCenter}{stgoorange}
    \colorlet{colorLabel}{stgored}
    \tikzstyle{nodeStyle}=[circle, draw, fill=white, inner sep=0pt,
      minimum size=5ex]
    \tikzstyle{neighbNodeStyle}=[circle, draw, fill=white, inner sep=1pt,
      minimum size=3cm]
    \tikzstyle{labelStyle}=[draw=none, minimum size=0ex, color=colorLabel]
  
    % Define ellipse parameters
    \def\radiusA{8cm}
    \def\radiusB{4cm}
    \def\radius{{\radiusA} and \radiusB}

    % Define positions
    \coordinate (u) at (180:\radius);
    \coordinate (v1) at (110:\radius);
    \coordinate (w1) at (55:\radius);
    \coordinate (v2) at (-110:\radius);
    \coordinate (w2) at (-55:\radius);

    % Draw neighborhoods
    \def\radiuss{5.5cm and 2cm}
    \def\radiusl{11cm and 6cm}
    \draw[draw=none,fill=stgoorange!10] (150:\radiuss) arc (150:210:\radiuss)
      -- (225:\radiusl) arc (225:125:\radiusl) -- cycle;
    \draw[draw=none,fill=stgogreen!10] (150:\radiuss) arc (150:0:\radiuss)
      -- (0:\radiusl) arc (0:135:\radiusl) -- cycle;
    \draw[draw=none,fill=stgoblue!10] (-150:\radiuss) arc (-150:0:\radiuss)
      -- (0:\radiusl) arc (0:-135:\radiusl) -- cycle;

    % Draw edges
    \draw (0,0) ellipse ({\radiusA} and \radiusB);

    % Draw nodes
    \def\distNeighb{2.5mm}
    \node[neighbNodeStyle] at (u) {\begin{tikzpicture}
      \node[nodeStyle,fill=colorCenter,label={[labelStyle]90:$\bot$}] 
        (x) at (0,0) {$1$};
      \node[nodeStyle,label={[labelStyle]90:$\top$}] 
        (y) [right = \distNeighb of x] {$2$};
      \draw (x) -- (y);
      \draw[draw=none] ($(x)!0.5!(y)$) circle (1cm);
    \end{tikzpicture}};
    \node[neighbNodeStyle] at (v1) {\begin{tikzpicture}
      \node[nodeStyle,fill=colorCenter,label={[labelStyle]90:$\top$}] 
        (x) at (0,0) {$2$};
      \node[nodeStyle,label={[labelStyle]90:$\bot$}] 
        (y1) [left = \distNeighb of x] {$1$};
      \node[nodeStyle,label={[labelStyle]90:$0$}] 
        (y2) [right = \distNeighb of x] {$3$};
      \draw (x) -- (y1);
      \draw (x) -- (y2);
      \draw[draw=none] (x) circle (1cm);
    \end{tikzpicture}};
    \node[neighbNodeStyle] at (v2) {\begin{tikzpicture}
      \node[nodeStyle,fill=colorCenter,label={[labelStyle]90:$\top$}] 
        (x) at (0,0) {$2$};
      \node[nodeStyle,label={[labelStyle]90:$\bot$}] 
        (y1) [left = \distNeighb of x] {$1$};
      \node[nodeStyle,label={[labelStyle]90:$1$}] 
        (y2) [right = \distNeighb of x] {$4$};
      \draw (x) -- (y1);
      \draw (x) -- (y2);
      \draw[draw=none] (x) circle (1cm);
    \end{tikzpicture}};
    \node[neighbNodeStyle] at (w1) {\begin{tikzpicture}
      \node[nodeStyle,fill=colorCenter,label={[labelStyle]90:$0$}] 
        (x) at (0,0) {$3$};
      \node[nodeStyle,label={[labelStyle]90:$\bot$}] 
        (y1) [left = \distNeighb of x] {$2$};
      \node[nodeStyle,label={[labelStyle]90:$1$}] 
        (y2) [right = \distNeighb of x] {$4$};
      \draw (x) -- (y1);
      \draw (x) -- (y2);
      \draw[draw=none] (x) circle (1cm);
    \end{tikzpicture}};
    \node[neighbNodeStyle] at (w2) {\begin{tikzpicture}
      \node[nodeStyle,fill=colorCenter,label={[labelStyle]90:$1$}] 
        (x) at (0,0) {$4$};
      \node[nodeStyle,label={[labelStyle]90:$\bot$}] 
        (y1) [left = \distNeighb of x] {$2$};
      \node[nodeStyle,label={[labelStyle]90:$0$}] 
        (y2) [right = \distNeighb of x] {$3$};
      \draw (x) -- (y1);
      \draw (x) -- (y2);
      \draw[draw=none] (x) circle (1cm);
    \end{tikzpicture}};
\end{tikzpicture}

\end{document}
