\usepackage{tcolorbox}% http://ctan.org/pkg/tcolorbox
\definecolor{framecolor}{rgb}{0.122, 0.435, 0.698}% Frame (left border) color
\definecolor{bgcolor}{rgb}{0.95, 0.95, 0.95}% Background color

\newcommand{\goalbox}[1]{%
  \begin{tcolorbox}[
    colframe=framecolor,       % Frame color (applies to the left border)
    colback=bgcolor,           % Background color
    boxrule=0pt,               % No border for other sides
    leftrule=3pt,              % Thickness of the left border
    arc=0pt,                   % Square edges
    left=2pt,                  % Padding on the left inside the box
    right=2pt,                 % Padding on the right inside the box
    top=2pt,                   % Padding above the text
    bottom=2pt,                % Padding below the text
    width=\linewidth,          % Full width
    before skip=4pt,           % Space before the box
    after skip=4pt             % Space after the box
  ]
    #1
  \end{tcolorbox}
}




% draw a frame around given text for goal-blocks
\newcommand{\framedtext}[1]{%
    \par%
    \noindent\fbox{%
        \parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{#1}%
    }%
}

% Define a new environment for the rule blocks
\newcommand{\ruleblock}[2]{
    \vspace{0.5em}
    \noindent
    \fbox{
        \parbox{\textwidth}{
            \textbf{Rule #1:} #2
        }
    }
    \vspace{0.5em}
}
