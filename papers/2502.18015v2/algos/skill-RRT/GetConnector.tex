\begin{algorithm}[H]
\caption{\texttt{GetConnector}}\label{algo:GetConnector}
\begin{algorithmic}[1]

\Procedure{\texttt{GetConnector}}{$K, \mathcal{K}=\{K_\text{P},K_{\text{NP}_1},\dots, K_{\text{NP}_n}\}, C=\{c_\text{P},c_{\text{NP}_1},\dots, c_{\text{NP}_n}\}$}
    \For{$K_\alpha \in \mathcal{K}$}
        \If{$K == K_\alpha$}
            \State \textbf{return} $c_\alpha$
        \EndIf
    \EndFor
\EndProcedure

\end{algorithmic}
\end{algorithm}
