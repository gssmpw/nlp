\begin{algorithm}[H]
\caption{\texttt{GetRegion}}\label{algo:GetRegion}
\begin{algorithmic}[1]

\Procedure{\texttt{GetRegion}}{$K, \mathcal{K},\{\mu^i\}_{i=1,\dots, n}$}
    \For{$i = 1$ to $n$}
        \If{$K == K_{\text{NP}_i}$}
            \State \textbf{return} $\mu^i$
        \EndIf
    \EndFor
    \State \textbf{return} $\cup^n_{i=1}\mu^i$
\EndProcedure

\end{algorithmic}
\end{algorithm}
