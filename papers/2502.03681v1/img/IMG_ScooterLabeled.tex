	% Input:
	%   #1 Line offset (optional)
	%   #2 Line angle
	%   #3 Line length
	%   #4 Line label
	% Example:
	%   \lineann[1]{30}{2}{$L_1$}
	\newcommand{\lineann}[4][0.5]{%
		\begin{scope}[rotate=#2, inner sep=2pt]
			\draw[dashed] (0,0) -- +(0,#1)
			node [coordinate, near end] (a) {};
			\draw[dashed] (#3,0) -- +(0,#1)
			node [coordinate, near end] (b) {};
			\draw[|<->|] (a) -- node[rotate = #2, fill=white] {#4} (b);
		\end{scope}
	}
