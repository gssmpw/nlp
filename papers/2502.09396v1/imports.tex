\newcommand{\cellcolorgrey}[1]{ % Map cell to grey shading
    \begingroup
    \edef\tempa{#1}
    \ifx\tempa\empty%
    \else
        \pgfmathsetmacro{\scaled}{max(0,min(100,round(100*abs(#1))))}
        \xdef\clr{gray!\scaled!white}
        \cellcolor{\clr}{#1}
    \fi
    \endgroup
}
