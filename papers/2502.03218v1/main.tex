\documentclass[10pt, twocolumn]{article}
\usepackage[utf8]{inputenc}
\usepackage[backend=biber,style=numeric,sorting=none]{biblatex}
\addbibresource{references.bib}
% Add these commands to suppress TeX-related metadata
%\pdfinfoomitdate=1
%\pdftrailerid{}
%\pdfsuppressptexinfo=1

% Include packages and configurations from separate files
\input{packages}       % Packages used in the document
\input{customization}  % Customizations like colors, header/footer settings

% Document starts here
\begin{document}

% Conditional one/two-column layout for title, authors, and abstract
\checktwocolumn{
    % If two-column, use this layout
    \twocolumn[{
        \maketitle

        % Abstract
        \input{abstract}   % Abstract section

       
    }]
}{
    % If one-column, use this layout
    \maketitle

    % Abstract
    \input{abstract}   % Abstract section

   
}

% Body content sections
\setlength{\parindent}{0pt}

\input{introduction}    % Introduction section
\input{lr}              % Literature Review section
\input{methods}         % Methods section
%\input{results}         % Results section
\input{discussion}      % Discussion section
\input{conclusion}      % Conclusion section

% Declaration and other sections
\input{declaration}

% References Section
\printbibliography[heading=bibintoc, title={References}]

\end{document}
