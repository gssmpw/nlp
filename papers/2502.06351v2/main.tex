% Command for mean +/- std
% Example usage: \ms{1.0}{0.1}[b] or \ms{1.0}{0.1}
\NewDocumentCommand{\ms}{m m o}{%
  \IfNoValueTF{#3}% Check if the third argument is provided
    {$#1{\scriptstyle \pm #2}$}% If no third argument (default case)
    {\ifthenelse{\equal{#3}{b}}% If third argument is "b", bold version
      {$\mathbf{#1}{\scriptstyle \mathbf{\pm #2}}$}% Bold version
      {\ifthenelse{\equal{#3}{u}}% If third argument is "u", underscored version
        {\underline{$#1{\scriptstyle \pm #2}$}}% Underscored version
        {$#1{\scriptstyle \pm #2}$}% Normal version
      }%
    }%
}
