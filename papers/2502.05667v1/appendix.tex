// the probability of collider appearing
const double pcollider; 
// when collider appearing, the chance of collision posture of collider
const double pocc; 

module ManagedComponents
    z : [0..4] init 0; 
    // 0:check collider, 1:collider detected,
    // 2:check wait, 3:no collider, 4:done
    [look] t=1 & z=0 -> 
    pcollider:(z'=1) + (1-pcollider):(z'=3);
    [check] t=1 & z=1 -> 1:(z'=2);
    [retry] t=1 & z=2 & wait -> 1:(z'=0);
    [proceed] t=1 & z=2 & !wait -> 1:(z'=3);
    [travel] t=1 & z=3 -> 1:(z'=4);
    [end] t=1 & z=4 -> 1:(z'=4);
endmodule

// these four parameters to abstract the DNN component's behavior
const double pClass0AsClass0;
const double pClass0AsClass1;
const double pClass1AsClass0;
const double pClass1AsClass1;

// model the peception behavior
module EnvironmentMonitor
    k2 : [1..2] init 1;
    //1:not on collision course (occ), 2:occ
    k : [1..2] init 1; 
    [monitor] t=2 -> (1-pocc)*pClass0AsClass0:(k'=1)&
    (k2'=1)+(1-pocc)*pClass0AsClass1:(k'=1)&
    (k2'=2)+pocc*pClass1AsClass0:(k'=2)&
    (k2'=1)+pocc*pClass1AsClass1:(k'=2)&(k2'=2);
endmodule

// prob. of moving when DNN predicts not occ
const double c1; 
// prob. of moving when DNN predicts occ
const double c2;

module Controller
    wait : bool init false;
    [decide] t=3 & k2=1 -> 
    (1-c1):(wait'=true) + c1:(wait'=false);
    [decide] t=3 & k2=2 -> 
    (1-c2):(wait'=true) + c2:(wait'=false);
endmodule

module Turn
    t : [1..3] init 1;
    [check] true -> 1:(t'=2);
    [monitor] true -> 1:(t'=3);
    [decide] true -> 1:(t'=1);
endmodule

// reward structure
rewards "time"
    [travel] true : 10; // the time of movement
    [proceed] k=2 : 3; // the penalty for collison
    [retry] true : 5; // the time of waiting
endrewards

label "collision" = z=3 & k=2;
label "done" = z=4;
\end{lstlisting}
