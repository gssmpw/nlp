\section{Basic Algorithm}

\begin{algorithm}[ht]
\caption{\basicalgo
 \newline  {\bf Input: } User sets $\sets = \{(u, S_u)\}_{u \in [n]}$
 \newline  {\bf Output: } $w: \U \to \mathbb{R}$ weighting of the items
}
\begin{algorithmic}[1]
\label{alg:basic}
\STATE Initialize weight vector $w$ with zeros
\FORALL{$u \in [n]$}
    \STATE $w(i) \pluseq 1/\sqrt{|S_u|}$ for $i \in S_u$ \COMMENT{Add basic $\ell_2$ bounded weight.}
\ENDFOR
\RETURN $w$
\end{algorithmic}
\end{algorithm}
