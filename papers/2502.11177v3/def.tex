\newcommand{\numedit}[1]{#1}

\newcommand{\numrwe}[2]{%
    \begin{tikzpicture}[baseline]
        \pgfmathparse{#1 > #2 ? 1 : 0}
        \ifnum\pgfmathresult=0
            \pgfmathsetmacro{\percentdiff}{min(130, 130*(#2-#1)/#2)}
            \pgfmathsetmacro{\intensity}{\percentdiff}
            \fill[monte_carlo!\intensity!white, rounded corners=1] (-0.6em, -0.22em) rectangle (2.6em, 0.92em);
        \fi
        \node[inner sep=0pt] at (1em, 0.7ex) {#1};
    \end{tikzpicture}%
}



\newcommand{\colorbar}{%
    \begin{tikzpicture}
    \shade[left color=white, right color=monte_carlo] (0,0) rectangle (10,0.5);
        
        \foreach \x in {0,0.1,...,1} {
            \draw[white, thick] (\x*10,-0.1) -- (\x*10, 0.1);
            \node[below] at (\x*10, 0) {\small \pgfmathprintnumber[fixed,precision=1]{\x}};
        }
    \end{tikzpicture}%
}

        }
