%% Note: Custom new commands for styling participant quotes.
\newcommand{\inlinequote}[1]{\emph{``#1''}\allowbreak} % Italic text in quotation marks

\newcommand{\tablequote}[2]{\emph{``#1''} (#2)\allowbreak} % Italic text in quotation marks

\newcommand{\blockquote}[2]{           % Italic block quote, no quotation marks or P#
    \begin{quote}\emph{``#1''} (#2)\end{quote}\allowbreak}

%% Note: Commands for custom styling of bold, inline paragraph headings
\newcommand{\parHeading}[1]{\vspace{2mm}\noindent{\textbf{#1.}\allowbreak}}

\newcommand{\ie}[1]{(i.e., {#1}\allowbreak)}

\newcommand{\eg}[1]{(e.g., {#1}\allowbreak)}

\newcommand{\entry}[1]{P$_{e}#1$}

\newcommand{\diary}[1]{P$_{d}#1$}

\newcommand{\codesign}[1]{P$_{c}#1$}
