\begin{algorithm}
\caption{\methodname{} Training}
\KwIn{Coarse-to-fine Autoencoder $\text{Enc}$, $\text{Dec}$}
\KwOut{}
\For{$i \gets 1$ \textbf{to} $n-1$}{
    \For{$j \gets 1$ \textbf{to} $n-i$}{
        \If{$L[j] > L[j+1]$}{
            Swap $L[j]$ and $L[j+1]$
        }
    }
}
\Return $L$
\end{algorithm}
