\section{Pseudo Code of URECA}
\begin{algorithm}
\caption{URECA}
\label{alg:algorithm}
\textbf{Input}:  Q, D, GT \\
\textbf{Output}: NONE
\begin{algorithmic}[1]
\State $E=Q \cdot D$
\State $DYN=softmax(Q \cdot Q)$
\For{$k \gets 1$ to $max\_recursion\_num$}
\State $S = argsort(E)$
\State $p = S.find\_position(GT)$
\State $F, B = E.split(p)$
\State $F = F + F \cdot DYM $
\State $E=F$
\EndFor
\end{algorithmic}
\end{algorithm}
