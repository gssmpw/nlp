\begin{tikzpicture}[
    node distance=2.5cm,
    state/.style={circle, draw, minimum size=1cm},
    action1/.style={-{Stealth[scale=1.2]}, thick, blue},
    action2/.style={-{Stealth[scale=1.2]}, thick, red, dashed},
    reset/.style={-{Stealth[scale=1.2]}, thick, green, bend right=40},
    reward/.style={above, midway, font=\small}
]

    % States
    \foreach \i in {0,1,...,12} {
        \node[state] (s\i) at (\i*1.5, 0) {$s_{\i}$};
    }

    % Transitions
    \foreach \i [evaluate=\i as \next using int(\i+1), evaluate=\i as \nextnext using int(\i+2)] in {0,1,...,10} {
        \draw[action1] (s\i) to node[reward] {$-3$} (s\next);
        \draw[action2] (s\i) to node[reward, below] {$-3$} (s\nextnext);
    }

    % Special transitions
    \draw[action1] (s11) to node[reward] {$-2$} (s12);
    \draw[reset] (s12) to node[reward, below] {$0$} (s0);

    % Policy annotations
    \node[above=0.2cm of s11, font=\small] {$\pi(s_{11}) = 1$};
    \node[above=0.2cm of s0, font=\small] {$\pi(s_{0-10}) \in \{1, 2\}$};

\end{tikzpicture}
