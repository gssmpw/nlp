% Define the grid size and separation
\def\rows{2} % Number of rows
\def\cols{3} % Number of columns

% Define the width and height of the image
\def\imagewidth{16} % Width of the image in cm
\def\imageheight{9} % Height of the image in cm
\def\sep{3} % Separation between patches in cm

 % Calculate patch dimensions
\pgfmathsetmacro{\patchwidth}{\imagewidth / \cols}
\pgfmathsetmacro{\patchheight}{(\imageheight / \rows}

\pgfmathsetmacro{\firstscore}{{
    {1, 2, 3},
    {4, 5, 6}
}}

% Draw the grid patches
\foreach \row in {0,...,\numexpr\rows-1} {
    \foreach \col in {0,...,\numexpr\cols-1} {
        % Calculate the position of each patch
        \pgfmathsetmacro{\xstart}{\col * (\patchwidth + \sep)}
        \pgfmathsetmacro{\ystart}{\imageheight - (\row + 1) * \patchheight - \row * \sep}

        % Clip the patch from the image
        \begin{scope}
            \clip (\xstart, \ystart) rectangle ++(\patchwidth, \patchheight);
            \node[anchor=south west, inner sep=0] at 
            (\col * \sep, - \row * \sep) {
                \includegraphics[width=\imagewidth cm, height=\imageheight cm]{figures/cats-1.png}
            };
        \end{scope}

        % First-order scores
        \draw[red, line width=5pt] (\xstart, \ystart) rectangle ++(\patchwidth, \patchheight);
    }
}



\end{tikzpicture}
