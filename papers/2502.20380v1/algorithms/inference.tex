\begin{algorithm}[tb]
   \caption{\method: Inference loop}
   \label{algo:pseudo_inference_mucode}
\begin{algorithmic}[1]
\INPUT Generator $\pi_\theta$, learned verifier~$R_{\phi}$, turn limit T, number of rollouts N, public tests, and private tests \\
\STATE Set $s_1=\{x\}$, $t=1$
% \INPUT Oracle verifier $R$ and $\mathcal{D}_0=\{\}$
\WHILE{true}
    \STATE Generate N rollouts $\{y^n_t\}_{n=1}^N \sim\pi_\theta(.|s_{t}) $ 
    \STATE Choose best solution $y_t^*=\arg \max_{n} R_{\phi}(x,y^{n}_t)$
    \STATE Execute $y_t^*$ to get execution feedback $o_t$
    \IF{$y_t^*$ passes public tests or $t = T$}
        \STATE break;
    \ENDIF
    \STATE Update state $s_{t+1}=\{s_t, y_t^*, o_t\}$ and increment $t$
\ENDWHILE
\OUTPUT Return $y^*$ to execute on public and private tests
\end{algorithmic}
\end{algorithm}
